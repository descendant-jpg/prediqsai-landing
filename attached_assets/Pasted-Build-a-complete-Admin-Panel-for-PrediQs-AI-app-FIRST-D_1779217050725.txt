CREATE TABLE IF NOT EXISTS 
admin_logs (
  id uuid primary key 
    default gen_random_uuid(),
  admin_email text,
  action text,
  target_user_id uuid,
  details text,
  created_at timestamptz 
    default now()
);

CREATE TABLE IF NOT EXISTS
app_config (
  key text primary key,
  value text,
  updated_at timestamptz 
    default now()
);

INSERT INTO app_config VALUES

ON CONFLICT (key) DO NOTHING;

UPDATE profiles 
SET is_admin = true 
WHERE email = 
  (SELECT value FROM app_config 
   WHERE key = 'admin_email'
   LIMIT 1);

THIRD: ADMIN ACCESS POINTS

3 ways to access admin panel:

1. Settings screen:
If profiles.is_admin = true
show button at bottom of settings:
"🔐 Admin Panel →"
Only visible to admin account
Tapping navigates to admin dashboard

2. Secret tap:
On settings screen tap app 
version number 7 times quickly
Shows password prompt
Enter ADMIN_PASSWORD
Grants admin access

3. URL route:
/admin or screen AdminPanel
Checks is_admin = true
Redirects non-admins away
