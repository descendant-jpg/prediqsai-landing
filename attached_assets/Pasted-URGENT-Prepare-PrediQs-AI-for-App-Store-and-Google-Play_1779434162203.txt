PRIORITY 7: BUILD FOR PRODUCTION

Prepare the app for production build:

1. Update app.json with:
{
  "expo": {
    "name": "PrediQs AI",
    "slug": "prediqsai",
    "version": "1.0.0",
    "orientation": "portrait",
    "icon": "./assets/icon.png",
    "splash": {
      "image": "./assets/splash.png",
      "resizeMode": "contain",
      "backgroundColor": "#070B12"
    },
    "ios": {
      "supportsTablet": false,
      "bundleIdentifier": 
        "com.prediqsai.app",
      "buildNumber": "1"
    },
    "android": {
      "package": "com.prediqsai.app",
      "versionCode": 1,
      "adaptiveIcon": {
        "foregroundImage": 
          "./assets/adaptive-icon.png",
        "backgroundColor": "#070B12"
      }
    }
  }
}

2. Create EAS build config (eas.json):
{
  "cli": {
    "version": ">= 5.9.1"
  },
  "build": {
    "development": {
      "developmentClient": true,
      "distribution": "internal"
    },
    "preview": {
      "distribution": "internal"
    },
    "production": {
      "autoIncrement": true
    }
  },
  "submit": {
    "production": {}
  }
}

3. Install EAS CLI:
npm install -g eas-cli
eas login
eas build:configure

4. Build for iOS:
eas build --platform ios 
  --profile production

5. Build for Android:
eas build --platform android 
  --profile production
